class Node {
  constructor(value) {
    this.value = value;
    this.next = null;
  }
}

class LinkedList {
    constructor()
    {
        this.head = null;
        this.size = 0;
        this.push = this.push.bind(this);
    }

    push(element) {
      const node = new Node (element);
      var current;
      if (!this.head) {
        this.head = node;
      } else {
        current = this.head;
        while(current.next) {
          current = current.next;
        }
        current.next = node;
        this.size++;
      }
    }

  let tempCurrentNode;
  while(currentNode){
    if (currentNode === this.head){
      uniqueList[0] = currentNode.value;

      this.removeNode(currentNode, previousNode);

      currentNode=null;
    }
  }
}


LinkedList.prototype.removeNode = function(currentNode, previousNode){
  previousNode.next = currentNode.next;
  delete currentNode.value;
  delete currentNode.next;
  this.size--;
}


LinkedList.prototype.isPresent = function(value, list) {
  for (let elementIndex = 0; elementIndex < list.length; elementIndex ++) {
    const element = list[elementIndex];
    if (element === value) {
      return true;
    }
  }
  return false;
}

LinkedList.prototype.push = function(element) {
  const node = new Node (element);
  var current;
  if (!this.head) {
    this.head = node;
    this.size++;
  } else {
    current = this.head;
    while(current.next) {
      current = current.next;
    }
    current.next = node;
    this.size++;
  }
}


LinkedList.prototype.pop = function(){
  if (!this.head.next){
    const poppedNode = this.head;
    this.head = this.head.next;
    this.size--;
    return poppedNode;
  } else {
    let previousNode = this.head;
    let currentNode = this.head.next
    while(currentNode.next){
      currentNode = currentNode.next;
      previousNode = previousNode.next;
    }
    const poppedNode = currentNode;
    previousNode.next = currentNode.next;
    this.size--;
    return poppedNode;
  }
}

function reverseList(list){
  var reversedList = new LinkedList;
  while(list.head){
    const item = list.pop().value;
    reversedList.push(item);
  }
  return reversedList;
}
